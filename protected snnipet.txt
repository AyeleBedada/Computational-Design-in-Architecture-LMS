<script>
  function requireSession() {
    const session = JSON.parse(localStorage.getItem('sessionUser')||'null');
    if (!session) location.href = 'index.html';
    return session;
  }
  function renderChrome(activeHref) {
    const s = requireSession();
    const role = s.role;

    // Top bar
    document.getElementById('topbar').innerHTML = `
      <div class="topbar-inner">
        <div class="logo">CDA</div>
        <nav class="topnav">
          <a href="00_navigation_tutorial.html">Tutorial</a>
          <a href="00_intro.html">Intro</a>
          <a href="Unit-0-[Course Introduction]/0_course_syllabus.html">Syllabus</a>
          <a href="complete.html">Completion</a>
        </nav>
        <div class="top-actions">
          <span class="userchip">${s.username} (${s.role})</span>
          ${role==='admin' ? '<a class="btn tiny" href="admin.html">Admin</a>' : ''}
          <button class="btn tiny outline" onclick="logout()">Logout</button>
        </div>
      </div>`;

    // Sidebar
    const links = [
      {href:'00_navigation_tutorial.html', label:'How to Navigate'},
      {href:'00_intro.html', label:'Course Introduction'},
      {href:'00_concern.html', label:'Concerns & Suggestions'},
      {href:'Unit-0-[Course Introduction]/0_unit_introduction.html', label:'Unit 0: Intro'},
      {href:'Unit-0-[Course Introduction]/0_course_syllabus.html', label:'Syllabus'},
      {href:'Unit-0-[Course Introduction]/0_rubrics.html', label:'Rubrics'},
      {href:'Unit-0-[Course Introduction]/0_pre_assessment.html', label:'Pre-Assessment'},
      {href:'Unit-0-[Course Introduction]/0_quiz_0.html', label:'Unit 0 Quiz'},
      {href:'Unit-1-[Computational Design in Architecture]/1.1_[Introduction to Computational Design]/1.1.1_introduction_Computational_Design.html', label:'1.1.1 Intro to CD'},
      {href:'Unit-1-[Computational Design in Architecture]/1.2_handles_Manipulators.html', label:'1.2 Handles & Manipulators'},
      {href:'Unit-1-[Computational Design in Architecture]/1.3_visual_Programming.html', label:'1.3 Visual Programming'},
      {href:'Unit-2-[Topological Design Process and Form Finding]/2.1_contemporary_Geometries.html', label:'2.1 Contemporary Geometries'},
      {href:'Unit-3-[Digital and analog fabrication Techniques]/3.1_digital_fabrication.html', label:'3.1 Digital Fabrication'},

    document.getElementById('sidebar').innerHTML = `
      <div class="sidebar-inner">
        ${links.map(l=>`<a class="${activeHref===l.href?'active':''}" href="${l.href}">${l.label}</a>`).join('')}
      </div>`;
  }

  // Anti-copy/screenshot (students only)
  function enforceStudentProtection() {
    const s = JSON.parse(localStorage.getItem('sessionUser')||'null');
    if (!s || s.role!=='student') return; // admins can copy/screenshot
    document.addEventListener('contextmenu', e=>e.preventDefault());
    document.addEventListener('keydown', (e)=>{
      // Block common copy / print-screen
      if (e.ctrlKey && (e.key.toLowerCase()==='c' || e.key.toLowerCase()==='p' || e.key.toLowerCase()==='s')) e.preventDefault();
      if (e.key==='PrintScreen') {
        e.preventDefault();
        // attempt to clear clipboard / mask screen
        const mask = document.createElement('div');
        mask.style.cssText='position:fixed;inset:0;background:#000;opacity:.8;z-index:99999';
        document.body.appendChild(mask);
        setTimeout(()=>mask.remove(), 600);
      }
    });
    document.body.classList.add('no-select');
  }
</script>
